<body>
  <div id="game-container">
    <div id="dialog-box">
      <div id="dialog-text">HO HO HO Tata Lélé, J'imagine que tu es prête à recevoir ton cadeau.</div>
      <div id="options" style="display: none;">
        <div class="option selected">1. Un massage près de Vannes (dans la limite de 60 € bien évidemment pour ne pas que les autres se sentent lésés, petite maligne).</div>
        <div class="option">2. Rien, j'ai déjà ce qu'il me faut.</div>
      </div>
      <div id="instructions">Appuyez sur ESPACE pour continuer</div>
    </div>
  </div>

  <script>
    const dialogText = document.getElementById('dialog-text');
    const options = document.getElementById('options');
    const instructions = document.getElementById('instructions');
    let gameState = 'intro';
    let selectedOption = 0;

    const dialogues = {
      intro: "HO HO HO Tata Lélé, J'imagine que tu es prête à recevoir ton cadeau.",
      question: "Que voudrais-tu pour Noël ?",
      results: [
        "Excellent choix, va voir ton Secret Santa (si tu sais qui c'est) pour lui donner la meilleure adresse de massage de Vannes. Et surtout, joyeux Noël !",
        "Ho ho ho, heureusement que tout le monde ne réagit pas comme toi, tu veux la mort du petit commerce ou quoi ? Mais bon, joyeux Noël quand même."
      ]
    };

    document.addEventListener('keydown', (event) => {
      event.preventDefault();
      
      if (event.code === 'Space') {
        if (gameState === 'intro') {
          gameState = 'question';
          dialogText.textContent = dialogues.question;
          options.style.display = 'block';
          instructions.textContent = "Utilisez ↑↓ pour choisir et ESPACE pour valider";
        } else if (gameState === 'question') {
          gameState = 'result';
          options.style.display = 'none';
          dialogText.textContent = dialogues.results[selectedOption];
          instructions.textContent = "Appuyez sur ESPACE pour terminer";
        } else if (gameState === 'result') {
          document.getElementById('dialog-box').style.display = 'none';
        }
      }
      
      if (gameState === 'question') {
        const optionElements = document.querySelectorAll('.option');
        if (event.code === 'ArrowUp' && selectedOption > 0) {
          optionElements[selectedOption].classList.remove('selected');
          selectedOption--;
          optionElements[selectedOption].classList.add('selected');
        } else if (event.code === 'ArrowDown' && selectedOption < 1) {
          optionElements[selectedOption].classList.remove('selected');
          selectedOption++;
          optionElements[selectedOption].classList.add('selected');
        }
      }
    });
  </script>
</body>
